function filterCategory(category, element) { currentCategory = category; // تحديث التنسيق النشط للأزرار document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active')); if (element) { element.classList.add('active'); } executeSearchAndFilter(); } // دالة البحث المدمجة مع الفلترة function executeSearchAndFilter() { const query = document.getElementById('searchInput').value.trim().toLowerCase(); const filtered = productsData.filter(p => { const matchesCategory = (currentCategory === 'all' || p.category === currentCategory); const matchesSearch = p.title.toLowerCase().includes(query); return matchesCategory && matchesSearch; }); displayProducts(filtered); } // الاستماع لمدخلات صندوق البحث document.getElementById('searchInput').addEventListener('input', executeSearchAndFilter); // فتح نافذة تفاصيل المنتج function openProduct(id) { const p = productsData.find(prod => prod.id === id); if (!p) return; currentOpenedProductId = id; const modal = document.getElementById('productModal'); const img = document.getElementById('modalImg'); const title = document.getElementById('modalTitle'); const variantContainer = document.getElementById('variantContainer'); const variantSelect = document.getElementById('variantSelect'); const desc = document.getElementById('modalDesc'); const actionBtn = document.getElementById('modalActionBtn'); // جلب رابط الصورة الفعلي أو الصورة البديلة في حال فشل التحميل const cardImg = document.getElementById(`img-prod-${id}`); img.src = cardImg ? cardImg.src : p.image; title.textContent = p.title; desc.textContent = p.desc; if (p.hasVariants) { variantContainer.style.display = 'block'; variantSelect.innerHTML = ''; p.variants.forEach((v, index) => { const opt = document.createElement('option'); opt.value = index; opt.textContent = `${v.name} - ${v.price.toFixed(2)} د.أ`; variantSelect.appendChild(opt); }); } else { variantContainer.style.display = 'none'; } updateModalPrice(); // ربط زر الإضافة في المودال بوظيفة الإضافة الفاصلة actionBtn.onclick = function() { addToCart(p.id); modal.classList.remove('active'); }; modal.classList.add('active'); } // تحديث السعر داخل النافذة المنبثقة عند تغيير الفئة function updateModalPrice() { const p = productsData.find(prod => prod.id === currentOpenedProductId); if (!p) return; const priceTag = document.getElementById('modalPrice'); if (p.hasVariants) { const select = document.getElementById('variantSelect'); const selectedIndex = select.value; const price = p.variants[selectedIndex].price; priceTag.textContent = `${price.toFixed(2)} د.أ`; } else { priceTag.textContent = `${p.price.toFixed(2)} د.أ`; } } // إغلاق المودال عند الضغط في المساحة الخارجية function closeModal(event) { if (event.target.id === 'productModal') { document.getElementById('productModal').classList.remove('active'); } } // التحكم في فتح وإغلاق قائمة السلة الجانبية function toggleCart() { const sidebar = document.getElementById('cartSidebar'); if (sidebar.style.left === '0px' || sidebar.classList.contains('open')) { sidebar.style.left = '-360px'; sidebar.classList.remove('open'); } else { sidebar.style.left = '0px'; sidebar.classList.add('open'); } } // إضافة المنتج إلى السلة function addToCart(id) { const p = productsData.find(prod => prod.id === id); if (!p) return; let itemName = p.title; let itemPrice = p.price; if (p.hasVariants) { const select = document.getElementById('variantSelect'); const selectedIndex = select.value; itemName = `${p.title} (${p.variants[selectedIndex].name})`; itemPrice = p.variants[selectedIndex].price; } // التحقق إذا كان المنتج بنفس الفئة مضاف مسبقاً لزيادة الكمية const existingItem = cart.find(item => item.name === itemName); if (existingItem) { existingItem.qty += 1; } else { cart.push({ id: p.id, name: itemName, price: itemPrice, qty: 1 }); } updateCartUI(); } // حذف عنصر من السلة function removeFromCart(index) { cart.splice(index, 1); updateCartUI(); } // تحديث واجهة السلة والعدادات function updateCartUI() { const cartCountLabel = document.getElementById('cartCount'); const cartItemsContainer = document.getElementById('cartItems'); const cartTotalLabel = document.getElementById('cartTotal'); cartItemsContainer.innerHTML = ''; let totalQty = 0; let totalPrice = 0; cart.forEach((item, index) => { totalQty += item.qty; totalPrice += (item.price * item.qty); const itemDiv = document.createElement('div'); itemDiv.className = 'cart-item'; itemDiv.innerHTML = `
${item.name}
${item.qty} × ${item.price.toFixed(2)} د.أ
`; cartItemsContainer.appendChild(itemDiv); }); cartCountLabel.textContent = totalQty; cartTotalLabel.textContent = `${totalPrice.toFixed(2)} د.أ`; } // إرسال الطلب المطور عبر الواتساب لتأكيد الشراء الفوري function checkout() { if (cart.length === 0) { alert('سلة المشتريات فارغة حالياً!'); return; } let message = `مرحباً Apricot Gamers، أريد تأكيد طلب الشراء التالي:\n\n`; let totalPrice = 0; cart.forEach((item, index) => { const subtotal = item.price * item.qty; totalPrice += subtotal; message += `${index + 1}. *${item.name}*\n الكمية: ${item.qty} | السعر: ${subtotal.toFixed(2)} د.أ\n`; }); message += `\n💰 *المجموع الكلي:* ${totalPrice.toFixed(2)} د.أ\n`; message += `💳 *طريقة الدفع المفضلة:* CliQ / محفظة إلكترونية`; // ترميز النص ليتوافق مع الروابط الإلكترونية const whatsappUrl = `https://wa.me/962796507200?text=${encodeURIComponent(message)}`; window.open(whatsappUrl, '_blank'); } // تعديل وظيفة زر تأكيد الطلب لتفعيل الدالة الجديدة بدلاً من الـ Alert العادي document.querySelector('.checkout-btn').setAttribute('onclick', 'checkout()'); // تشغيل العرض الأولي للمنتجات عند تحميل الصفحة لأول مرة window.onload = function() { displayProducts(productsData); };
